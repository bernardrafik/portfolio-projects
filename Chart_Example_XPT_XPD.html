import pandas as pd 
import matplotlib.pyplot as plt
#wczytanie pliku platyna
df1 = pd.read_csv ("C:/Users/Dell/Desktop/Nauka_DA/Python/Chart_XPT_XPD/XPTUSD1440.csv", header=None, names=["Data", "Time","Open", "High", "Low", "Price", "Volume"])

# stworzenie jednej kolumny Date_time z dwóch kolumn 
df1["Date_Time"] = pd.to_datetime(df1["Data"] + " " + df1["Time"])

# usunięcie starych kolumn Data i Time
df1 = df1.drop(columns=["Data", "Time"])

#ustawienie Datetime jako indeks
df1 = df1.set_index("Date_Time")

# ustawienie Datetime jako indeks 
df1 = df1.drop(columns=["Open", "High", "Low"])

###################################################################################################

#wczytanie pliku pallad
df2 = pd.read_csv ("C:/Users/Dell/Desktop/Nauka_DA/Python/Chart_XPT_XPD/XPDUSD1440.csv", header=None, names=["Data", "Time","Open", "High", "Low", "Price", "Volume"])

# stworzenie jednej kolumny Datetime
df2["Date_Time"] = pd.to_datetime(df2["Data"] + " " + df2["Time"])

# usunięcie starych kolumn Data i Time
df2 = df2.drop(columns=["Data", "Time"])

#ustawienie Datetime jako indeks
df2 = df2.set_index("Date_Time")

# ustawienie Datetime jako indeks 
df2 = df2.drop(columns=["Open", "High", "Low"])

# rysowanie wykresu liniowego 
plt.figure(figsize=(12,6))
plt.plot(df1.index, df1["Price"], label="Platyna (XPTUSD)", linewidth=1)
plt.plot(df2.index, df2["Price"], label="Pallad (XPDUSD)", linewidth=1)

plt.title("Wykres liniowy XPTUSD")
plt.xlabel("Data")
plt.ylabel("Cena [USD]")
plt.legend()
plt.grid(True)
plt.show()
